# Get fila especifica
python main.py get usuarios user1

# Escanear toda la tabla
python main.py scan --tabla usuarios

# Escanear desde una fila específica
python main.py scan --tabla usuarios --start_row user2

# Escanear hasta una fila específica
python main.py scan --tabla usuarios --stop_row user2

# Escanear un rango de filas
python main.py scan --tabla usuarios --start_row user1 --stop_row user3

# Listar tablas
python main.py list

# Count rows
python main.py count usuarios

# Describe
python main.py describe usuarios 

# Put 
python main.py put usuarios user2 info_personal:nombre "Juan Esteban"

# disable: Deshabilita una tabla especificada
python main.py disable usuarios

# is_disable
python main.py is_disable usuarios

# enable
python main.py enable usuarios 

# is_enabled
python main.py is_enabled usuarios

# drop
python main.py drop usuarios

#alter_table
#Para añadir
python main.py alter usuarios testfamily --add
#Para modificar
python main.py alter --tabla usuarios --column_family testfamily --modify max_versions 5 ttl 1209600
#Para borrar
python main.py alter --tabla usuarios --column_family testfamily  --delete

Truncate 
python main.py truncate 'usuarios'
